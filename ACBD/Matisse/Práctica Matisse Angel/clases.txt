
create class Zona(
	id INTEGER,
 	nombre STRING,
 	valorMetro INTEGER,
 	indiceRevalorizacion INTEGER
 );

create class Vivienda(
	id INTEGER,
 	metros INTEGER,
 	direccion STRING,
 	piso INTEGER,
 	letra STRING
 );
 
create class Cliente(
	id INTEGER,
 	DNI STRING,
 	nombre STRING,
 	apellidos STRING
 );

create class Ventana(
	id INTEGER,
 	modelo STRING,
 	metros INTEGER,
 	orientacion STRING,
 	climalite BOOLEAN
 );


ALTER CLASS Zona
	ADD RELATIONSHIP contiene
		RELATIONSHIP SET (Vivienda)
 	INVERSE Vivienda.estaEn;

 
ALTER CLASS Vivienda
	ADD RELATIONSHIP estaEn 
		RELATIONSHIP SET (Zona)
			CARDINALITY (0, 1)
	INVERSE Zona.contiene;

 
ALTER CLASS Vivienda
	ADD RELATIONSHIP posiblePropietario 
		RELATIONSHIP SET (Cliente)
	INVERSE Cliente.quiereComprar; 
	
ALTER CLASS Cliente
	ADD RELATIONSHIP quiereComprar
		RELATIONSHIP SET (Vivienda)
 	INVERSE Vivienda.posiblePropietario;
 

ALTER CLASS Vivienda
	ADD RELATIONSHIP tieneVentana 
		RELATIONSHIP SET (Ventana)
	INVERSE Ventana.colocadaEn; 
	
ALTER CLASS Ventana
	ADD RELATIONSHIP colocadaEn
		RELATIONSHIP SET (Vivienda)
 	INVERSE Vivienda.tieneVentana;
